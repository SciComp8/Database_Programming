# Data types for MySQL
## Numeric data types for integers
- TINYINT(size)
- SMALLINT(size)
- MEDIUMINT(size)
- INT(size)
- BIGINT(size)

## Numeric data types for decimals
- FLOAT(size,d)
- DOUBLE(size,d)
- DECIMAL(size,d)

## Text data types
- CHAR(size)
- VARCHAR(size)
- TINYTEXT
- TEXT
- MEDIUMTEXT
- LONGTEXT
- BLOB
- MEDIUMBLOB
- LONGBLOB
- ENUM(x,y,z,etc.)
- SET

## Date/time data types
- DATE()
- DATETIME()
- TIMESTAMP()
- TIME()
- YEAR()
